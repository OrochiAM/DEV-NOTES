import session from 'express-session'

const secret = process.env.SPIRAL_SESSION_SECRET || 'jellyfish-baskingshark'

app.use(session({
  secret: secret,
  resave: false, - ne cuva se ako se nije promenila sesija
  saveUninitialized: false, - ne prave se za guest sesije
  cookie: {
    httpOnly: true, - ne moze da se accessuje preko js-a
    secure: false, - https 
    sameSite: 'lax' - kako se salju kolacici preko cross site zahteva (ne moze POST sa drugog sajta)
  }
}))

Protected Routes

Ovaj middleware se stavlja na rutu samo 
Custom middleware:

export function requireAuth(req, res, next) {

   if(!req.session.userId) {
      console.log('Acces blocked.')
      return res.status(401).json({ error: 'Unauthorized' })
   }

   next()
}

Primena:
app.use('/api/cart', requireAuth, cartRouter) - prvo odradi auth pa ide na next ako je sve kako treba
